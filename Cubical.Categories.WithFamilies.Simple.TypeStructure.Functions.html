private
  variable
    ℓC ℓC' ℓT ℓT' ℓD ℓD' ℓS ℓS' : Level

module _ ((C , Ty , Tm , term , ext) : SCwF ℓC ℓC' ℓT ℓT') where
  FunType : (A B : Ty) → Type _
  FunType A B =
    Σ[ A⇒B ∈ Ty ]
    PshIso (Tm A⇒B) (((Tm A) , λ Γ → ext Γ A) ⇒PshSmall Tm B)

  FunTypes : Type _
  FunTypes = ∀ A B → FunType A B
